cmake_minimum_required(VERSION 4.0)
if (APPLE)
    set(CMAKE_OSX_DEPLOYMENT_TARGET 26.0)
endif()
project(learn_vulkan LANGUAGES C CXX VERSION 1.0.0)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(TARGET vulkanApp)
set(TARGET_DIR ${PROJECT_SOURCE_DIR}/target)

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${TARGET_DIR}/bin)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${TARGET_DIR}/lib)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${TARGET_DIR}/lib)

add_executable(${TARGET})

find_package(Vulkan REQUIRED)
add_subdirectory(3rdparty)
add_subdirectory(src)

add_subdirectory(shaders)

set(ASSETS_SOURCE_DIR "${PROJECT_SOURCE_DIR}/assets")
set(ASSETS_BINARY_DIR "${PROJECT_SOURCE_DIR}/target/assets")
add_custom_target(copy_assets ALL
    COMMAND ${CMAKE_COMMAND} -E make_directory ${ASSETS_BINARY_DIR}
    COMMAND ${CMAKE_COMMAND} -E copy_directory
        ${ASSETS_SOURCE_DIR}
        ${ASSETS_BINARY_DIR}
    DEPENDS ${ASSETS_SOURCE_DIR}
    COMMENT "Copying assets directory to build directory"
)

add_dependencies(${TARGET} copy_assets)
